    Create UDF for histogram (2/2)
    %spark
    // create udf for histogram
    def udfMapToBin(valMin:Double, valMax:Double, numOfBins:Int) = {
    // create udf to map value into hist bin
    val binSize = (valMax - valMin) / numOfBins
    // udf
    def mapRuntimeToBin(runtime:Double) = {
    val lower = (runtime / binSize).toInt * binSize
    val upper = ((runtime / binSize).toInt + 1) * binSize
    BinStruct(f"$lower%.1f-$upper%.1f", lower, upper)
    }
    udf(mapRuntimeToBin _)
    }

    Compute Histogram
    %spark
    import org.apache.spark.sql.{DataFrame, Column}
    def histogram(df:DataFrame, column:Column, minVal:Double, maxVal:Double, numBins:Int) = {
    // aggregate into hist
    df
    .filter(column >= minVal && column <= maxVal)
    .withColumn("bin", udfMapToBin(minVal, maxVal, numBins)(column))
    .groupBy("bin")
    .count
    .orderBy(asc("bin.lower"))
    }

    Average movie rating by year
    %spark
    // actor performance evaluation
    // principalDf -> list of movie per (principal, year) -> (rating, numVotes)
    // -> avg(rating), sum(numVotes) -> accumulated avg(rating), sum(numVotes)
    // sumOfRating, totalVotes
    val getRating = udf((tid:String)=>
    bRatingMap.value.getOrElse(tid, (0.0, 0L)))
    val ratingReducer = udf((acc:(Double, Long), x:(Double,Long))=>{
    (acc._1+x._1*x._2, acc._2+x._2)
    })
    val scoredMoviePrincipalDf = moviePrincipalDf
    .groupBy("nconst", "startYear")
    .agg(collect_set("tconst").as("tconst_list"))
    .withColumn("rating_list", transform(col("tconst_list"), (v)=>getRating(v)))
    .withColumn("avg_score",
    aggregate(
    col("rating_list"),
    typedLit((0.0, 0L)),
    (acc, x)=>ratingReducer(acc, x)
    )
    )
    .withColumn("avg_score", $"avg_score".getField("_1") / $"avg_score".getField("_2"))
